## <a id="importing"></a>Import e utilização

```ts
// Adicione o modulo do Freud Select Button
import { FreudSelectButtonModule } from '@freud-ds/web-components';

// No Componente declare a lista de opções e o item selecionado por padrão
export class Componente {
    items: any[];
    selectedItem: any;
    constructor() {
        this.items = [
            {name: 'New York', code: 'NY'},
            {name: 'Rome', code: 'RM'},
            {name: 'London', code: 'LDN'},
            {name: 'Istanbul', code: 'IST'},
            {name: 'Paris', code: 'PRS'}
        ];
    }
}
```
