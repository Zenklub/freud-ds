## Getting Started
FreudConfirmDialog é definido usando a tag freud-confirm-dialog e uma instância de FreudConfirmationService é necessária para exibi-lo chamando o método confirm.
```html
<freud-confirm-dialog></freud-confirm-dialog>
<freud-button (click)="confirm()">Clique aqui!</freud-button>
```
```ts
import { FreudConfirmationService  } from '@freud-ds/web-components';

export class ConfirmDialogDemo {

    constructor(private confirmationService: FreudConfirmationService) {}

    confirm() {
        this.confirmationService.confirm({
            header: 'Confirme',
            message: 'Mensagem.',
            acceptLabel: 'Aceitar',
            rejectLabel: 'Rejeitar',
            accept: () => {
                //Ação quando o usuário aceitar
            },
            reject: () => {
                // Ação quando o usuário rejeitar
            }
        });
    }
}
```
