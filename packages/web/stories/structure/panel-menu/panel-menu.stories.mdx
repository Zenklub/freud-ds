## Getting Started
FreudPanelMenu requer uma coleção de itens de menu, para isso utilize a interface `FreudMenuItem`
```ts
import { FreudMenuItem } from '@freud-ds/web-components';

items: FreudMenuItem[] = [
    {
    label: 'File',
    icon: 'freud-icon freud-icon-pw pi-file',
    items: [{
      label: 'New',
      icon: 'freud-icon freud-icon-fw pi-plus',
      items: [
        {
            label: 'User',
            icon: 'freud-icon freud-icon-fw pi-user-plus',
            command: () => { /* Ação quando for clicado na opção */}
        },
        {
            label: 'Filter',
            icon: 'freud-icon freud-icon-fw pi-filter'
        }
      ]
    },
      {label: 'Open', icon: 'freud-icon freud-icon-fw pi-external-link'},
      {separator: true},
      {label: 'Quit', icon: 'freud-icon freud-icon-fw pi-times'}
    ]
  }
]
```
```html
      <freud-panel-menu
          [styleParams]="{'width':'400px'}"
          [items]="items">
</freud-panel-menu>
```
