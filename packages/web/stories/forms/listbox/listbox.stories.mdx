## Getting Started
Listbox requer um valor para vincular e uma coleção de opções.
```html
<freud-listbox [options]="cities" [(ngModel)]="selectedCity" optionLabel="name"></freud-listbox>
```

```ts
interface City {
    name: string,
    code: string
}

export class ListboxDemo {

    cities: City[];

    selectedCity: City;

    constructor() {
        this.cities = [
            {name: 'New York', code: 'NY'},
            {name: 'Rome', code: 'RM'},
            {name: 'London', code: 'LDN'},
            {name: 'Istanbul', code: 'IST'},
            {name: 'Paris', code: 'PRS'}
        ];
    }

}
```
